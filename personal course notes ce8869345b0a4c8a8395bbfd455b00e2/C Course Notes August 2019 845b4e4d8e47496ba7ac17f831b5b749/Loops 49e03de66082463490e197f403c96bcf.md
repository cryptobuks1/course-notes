Break
Useful to avoid having nested loops

```
// NESTED
if(condition){
	if(other condition){
		if(otherothercondition){
		    // code
		}
	}
}

public void DoThis(){
	// USING BREAK
	if(!condition)
		break;
	if(!othercondition)
		break;
	if(!otherothercondition)
		break;
	// code
}
```
